#include<stdio.h>
void swap(int *a,int *b){
    int c=*a;
    *a=*b;
    *b=c;
}
int get_pivot(int arr[],int l,int r){
    int pi_idx=r;
    int k=l-1;
    int i=l;
    while(i<r){
        if(arr[i]<arr[pi_idx]){
            
            k++;
            swap(&arr[k],&arr[i]);
        }
        i++;
    }
    k++;
    swap(&arr[k],&arr[r]);
    return k;
}
void quicksort(int arr[],int l,int r){
    if(l>=r) return;
    int pivot=get_pivot(arr,l,r);
    quicksort(arr,l,pivot-1);
    quicksort(arr, pivot+1, r);
}
int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    quicksort(arr, 0, n-1);
    for(int i=0;i<n;i++){
        printf("%d ",arr[i]);
    }
    return 0;
}
