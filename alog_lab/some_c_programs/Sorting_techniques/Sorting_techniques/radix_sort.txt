#include<stdio.h>
void swap(int *a,int *b){
    int c=*a;
    *a=*b;
    *b=c;
}
void count_sort(int arr[],int n,int num){
    int pre[10];
    for(int i=0;i<10;i++){
        pre[i]=0;
    }
    int res[n];
    for(int i=0;i<n;i++){
        res[i]=0;
    }
    for(int i=0;i<n;i++){
        pre[(arr[i]/num)%10]++;
    }
    for(int i=1;i<10;i++){
        pre[i]+=pre[i-1];
    }
    for(int i=n-1;i>=0;i--){
        res[pre[(arr[i]/num)%10]-1]=arr[i];
        pre[(arr[i]/num)%10]--;
    }
    for(int i=0;i<n;i++){
        arr[i]=res[i];
    }
    
}
void radixsort(int arr[],int n){
    int num=1;
    for(int pow=0;pow<3;pow++){
        
        count_sort(arr,n,num);
        num=num*10;
    }
    for(int i=0;i<n;i++){
        printf("%d ",arr[i]);
    }
}
int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    radixsort(arr, n);
    
    return 0;
}
