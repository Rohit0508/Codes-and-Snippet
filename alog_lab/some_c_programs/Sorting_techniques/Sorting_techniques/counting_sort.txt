#include<stdio.h>
void swap(int *a,int *b){
    int c=*a;
    *a=*b;
    *b=c;
}

int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    int pre[10];
    for(int i=0;i<10;i++) pre[i]=0;
    for(int i=0;i<n;i++){
        pre[arr[i]]++;
    }
    for(int i=1;i<10;i++){
        pre[i]=pre[i-1]+pre[i];
    }
    int res[n];
    for(int i=n-1;i>=0;i--){
        res[pre[arr[i]]-1]=arr[i];
        pre[arr[i]]--;
    }
    for(int i=0;i<n;i++){
        printf("%d ",res[i]);
    }
    return 0;
}
