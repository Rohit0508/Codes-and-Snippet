#include<stdio.h>
void swap(int *a,int *b){
    int c=*a;
    *a=*b;
    *b=c;
}
int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    
    int i=1;
    while(i<n){
        int j=i-1;
        int key=arr[j+1];
        while(j>=0&&arr[j]>=key){
            // printf("yes");
            arr[j+1]=arr[j];
            j--;
        }
        arr[j+1]=key;
        i++;
    }
    for(int k=0;k<n;k++){
        printf("%d ",arr[k]);
    }
    
    
    
    
    
    return 0;
}
