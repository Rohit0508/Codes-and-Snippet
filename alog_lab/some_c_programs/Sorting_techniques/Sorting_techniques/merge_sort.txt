#include<stdio.h>
void swap(int *a,int *b){
    int c=*a;
    *a=*b;
    *b=c;
}
void merge_array(int arr[],int l,int mid,int h){
    int n1=mid-l+1;
    int n2=h-mid;
    int nums1[n1];
    int k=0;
    for(int j=l;j<=mid;j++){
        nums1[k++]=arr[j];
    }
    k=0;
    int nums2[n2];
    for(int j=mid+1;j<=h;j++){
        nums2[k++]=arr[j];
    }
    // printf("yes \n");
    k=l-1;
    int i=0,j=0;
    while(i<n1&&j<n2){
        if(nums1[i]>=nums2[j]){
            arr[++k]=nums2[j];
            j++;
        }
        else{
            arr[++k]=nums1[i];
            i++;
        }
    }
    while(i<n1){
        arr[++k]=nums1[i++];
    }
    while(j<n2){
        arr[++k]=nums2[j++];
    }
}
void merge_sort(int arr[],int l,int h){
    if(l>=h) return;
    int mid=l+(h-l)/2;
    merge_sort(arr,l, mid);
    merge_sort(arr,mid+1,h);
    merge_array(arr,l,mid,h);
}
int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    merge_sort(arr, 0, n-1);
    for(int i=0;i<n;i++){
        printf("%d ",arr[i]);
    }
    return 0;
}
