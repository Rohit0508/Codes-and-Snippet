#include<stdio.h>
void swap(int *a,int *b){
    int c=*a;
    *a=*b;
    *b=c;
}
int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    
    for(int i=0;i<n;i++){
        int mini=i;
        for(int j=i+1;j<n;j++){
            if(arr[j]<arr[mini]){
                mini=j;
            }
        }
        // printf("%d \n",mini);
        swap(&arr[i],&arr[mini]);
    }
    for(int i=0;i<n;i++){
        printf("%d ",arr[i]);
    }
    
    
    
    return 0;
}
